13template <typename scalar>

15void IdealGasPhysics<scalar>::getDirectionalFlux(const scalar *const uc, const scalar *const n,
16 const scalar vn, const scalar p,
17 scalar *const __restrict flux) const
18{
19 flux[0] = vn*uc[0];
20 for(int i = 1; i < NDIM+1; i++)

22 flux[NDIM+1] = vn*(uc[NDIM+1] + p);

25template <typename scalar>
26inline
27void IdealGasPhysics<scalar>::getVarsFromConserved(const scalar *const uc, const scalar *const n,
28 scalar *const __restrict v,
29 scalar& vn,
30 scalar& p, scalar& H ) const
31{
32 for(int j = 0; j < NDIM; j++)
33 v[j] = uc[j+1]/uc[0];
34 vn = dimDotProduct(v,n);
35 const scalar vmag2 = dimDotProduct(v,v);
36 p = (g-1.0)*(uc[3] - 0.5*uc[0]*vmag2);
37 H = (uc[3]+p)/uc[0];
38}

65template <typename scalar>
66inline
67scalar IdealGasPhysics<scalar>::getDeltaPressureFromConserved(const scalar *const u,
68 const scalar *const du) const
69{
70 scalar unew[nvars];
71 for(int i = 0; i < nvars; i++)
72 unew[i] = u[i] + du[i];
73
74 scalar dp = 0;
75 for(int i = 2; i < NDIM+2; i++) {
76 dp -= ((u[i]+unew[i])*(u[0]+unew[0])/2.0*du[i]
77 - (unew[i]*unew[i]+u[i]*u[i])/2.0*du[0]);
78 }
79 dp = (g-1.0)*(du[nvars-1] - 1.0/(2*u[0]*unew[0])*dp);
80 return dp;
81}

374template <typename scalar>
375inline
376void IdealGasPhysics<scalar>::getGradPrimitive2FromConservedAndGradConserved(
377 const scalar *const __restrict uc, const scalar *const guc, scalar *const gup) const
378{
379 gup[0] = guc[0];
380
381 // velocity derivatives from momentum derivatives
382 for(int i = 1; i < NDIM+1; i++)
383 gup[i] = (guc[i]*uc[0]-uc[i]*guc[0])/(uc[0]*uc[0]);
384
385 const scalar p = getPressureFromConserved(uc);
386
387 /*
388 * Note that beyond this point, we assume we don't have access to momentum grads anymore,
389 * because we want to allow aliasing of guc and gp and we have modified gp above.
390 * So we only use density grad (gup[0] or guc[0]), velocity grads just computed (gup[1:NDIM+1])
391 * and energy grad (guc[NDIM+1]) going forward. We CANNOT use guc[1:NDIM+1].
392 */
393
394 // pressure derivative
395 scalar term1 = 0, term2 = 0;
396 for(int i = 1; i < NDIM+1; i++)
397 {
398 term1 += uc[i]*gup[i];
399 term2 += uc[i]*uc[i];
400 }
401 term2 *= 0.5*gup[0]/(uc[0]*uc[0]);
402 const scalar dp = (g-1.0)*(guc[NVARS-1] -term2 -term1);
403
404 // temperature
405 gup[NVARS-1] = getGradTemperature(uc[0],gup[0], p, dp);
406}

423template <typename scalar>
424inline
425void IdealGasPhysics<scalar>::getJacobianSutherlandViscosityWrtConserved(const scalar *const uc,
426 scalar *const __restrict dmu) const
427{
428 const scalar T = getTemperatureFromConserved(uc);
429 scalar dT[NVARS];
430 for(int i = 0; i < NVARS; i++)
431 dT[i] = 0;
432 getJacobianTemperatureWrtConserved(uc, dT);
433
434 const scalar coef = (1.0+sC/Tinf)/Reinf;
435 const scalar T15 = pow(T,1.5), Tm15 = pow(T,-1.5);
436 const scalar denom = (T + sC/Tinf)*(T+sC/Tinf);
437 for(int i = 0; i < NVARS; i++)
438 dmu[i] += coef* (1.5*Tm15*dT[i]*(T+sC/Tinf) - T15*dT[i])/denom;
439}

469template <typename scalar>
470inline
471void IdealGasPhysics<scalar>::getStressTensor(const scalar mu, const scalar grad[NDIM][NVARS],
472 scalar stress[NDIM][NDIM]) const
473{
474 // divergence of velocity times second viscosity
475 scalar ldiv = 0;
476 for(int j = 0; j < NDIM; j++)
477 ldiv += grad[j][j+1];
478 ldiv *= 2.0/3.0*mu;
479
480 for(int i = 0; i < NDIM; i++)
481 {
482 for(int j = 0; j < NDIM; j++)
483 stress[i][j] = mu*(grad[i][j+1] + grad[j][i+1]);
484
485 stress[i][i] -= ldiv;
486 }
487}
