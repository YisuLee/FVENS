39template <typename scalar>
40inline void getComponentsCartesian(const scalar mag, const scalar dir[NDIM], scalar vec[NDIM])
41{
42 zeros(vec,NDIM);
43 scalar cosphi, sinphi;
44
45 #ifndef USE_ADOLC
46 cosphi = (NDIM == 3) ? sqrt(dir[0]*dir[0]+dir[1]*dir[1]) : 1.0;
47 sinphi = (NDIM == 3) ? dir[2] : 0.0;
48 if(NDIM == 3) vec[2] = mag*sinphi;
49 #else
50 scalar one = 1.0;
51 scalar zero = 0.0;
52 scalar b = NDIM - 2;
53
54 condassign(cosphi, b, sqrt(dir[0]*dir[0]+dir[1]*dir[1]), one);
55 condassign(sinphi, b, dir[2], zero);
56 condassign(vec[2], b, mag*sinphi, zero);
57 #endif
58 vec[0] = mag*cosphi*dir[0];
59 vec[1] = mag*cosphi*dir[1];
60}
