159template <typename scalar, typename j_real = freal>
160class RoeAverageBasedFlux : public InviscidFlux<scalar,j_real>
161{
162public:
163 RoeAverageBasedFlux(const IdealGasPhysics<scalar> *const analyticalflux);
164 virtual void get_flux(const scalar *const ul, const scalar *const ur, const scalar* const n,
165 scalar *const flux) const = 0;
166 virtual void get_jacobian(const j_real *const ul, const j_real *const ur, const j_real* const n,
167 j_real *const dfdl, j_real *const dfdr) const = 0;
168
169protected:
170 using InviscidFlux<scalar,j_real>::physics;
171 using InviscidFlux<scalar,j_real>::jphy;
172 using InviscidFlux<scalar,j_real>::g;
173
175 void getRoeAverages(const scalar ul[NVARS], const scalar ur[NVARS], const scalar n[NDIM],
176 const scalar vi[NDIM], const scalar Hi,
177 const scalar vj[NDIM], const scalar Hj,
178 scalar& Rij, scalar& rhoij, scalar vij[NDIM],
179 scalar &vm2ij, scalar& vnij, scalar& Hij, scalar& cij) const
180 {
181 Rij = sqrt(ur[0]/ul[0]);
182 rhoij = Rij*ul[0];
183 for(int i = 0; i < NDIM; i++)
184 vij[i] = (Rij*vj[i] + vi[i])/(Rij + 1.0);
185 Hij = (Rij*Hj + Hi)/(Rij + 1.0);
186 vm2ij = dimDotProduct(vij,vij);
187 vnij = dimDotProduct(vij,n);
188 cij = sqrt( (g-1.0)*(Hij - vm2ij*0.5) );
189 }
190
192
194 void getJacobiansRoeAveragesWrtConserved(
195 const j_real ul[NVARS], const j_real ur[NVARS], const j_real n[NDIM],
196 const j_real vxi, const j_real vyi, const j_real Hi,
197 const j_real vxj, const j_real vyj, const j_real Hj,
198 const j_real dvxi[NVARS], const j_real dvyi[NVARS], const j_real dHi[NVARS],
199 const j_real dvxj[NVARS], const j_real dvyj[NVARS], const j_real dHj[NVARS],
200 j_real dRiji[NVARS], j_real drhoiji[NVARS], j_real dvxiji[NVARS], j_real dvyiji[NVARS],
201 j_real dvm2iji[NVARS], j_real dvniji[NVARS], j_real dHiji[NVARS], j_real dciji[NVARS],
